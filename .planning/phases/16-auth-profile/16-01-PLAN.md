    export function GoogleLoginButton({ onGoogleLogin }: GoogleLoginButtonProps) {

  <action>
    **Fix the logout socket disconnect bug in useAuth.ts:**

    The current `handleLogout` function calls `setSocketAuth(null)` but does NOT disconnect the socket. This means the socket retains its authenticated `socket.data.user` until the next natural reconnection.

    Current code (broken):
    ```typescript
    const handleLogout = async () => {
      if (token) {
        await apiCall('/logout', {
          method: 'POST',
          headers: { Authorization: `Bearer ${token}` },
        }).catch(() => {})
      }
      logout()
      setSocketAuth(null)
    }
    ```

    Replace with (fixed):
    ```typescript
    const handleLogout = async () => {
      if (token) {
        await apiCall('/logout', {
          method: 'POST',
          headers: { Authorization: `Bearer ${token}` },
        }).catch(() => {})
      }
      logout()
      setSocketAuth(null)
      // Reconnect socket without auth token so server sees guest identity
      if (socket.connected) {
        socket.disconnect()
        socket.connect()
      }
    }
    ```

    The import for `socket` is already present at the top of the file: `import { setSocketAuth, socket } from '../socket'`. No new imports needed.

    **Why this matters:** Without this fix, after logout the socket keeps its old authenticated identity. The server's `authMiddleware.ts` will still see the user as authenticated until the socket naturally disconnects. This breaks the guest-to-auth transition expected by AUTH-02.

    **No other changes.** AUTH-02 (guest mode) and AUTH-03 (JWT persistence) are already implemented:
    - AUTH-02: Socket auth middleware falls through to guest when no token. useAuth sets `loading: false` immediately if no saved token. Guest flow untouched.
    - AUTH-03: `authStore.setAuth()` saves token to localStorage. `useAuth` useEffect on mount calls `/api/auth/me` with saved token. Already working.
